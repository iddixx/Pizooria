Покупець => {
    Зробити покупця;
    Зробити чайові;
    Зробити чергу покупців;
    Зробити пропуск покупця;
    Зробити вкладку з покупцями;
} -> (Viktor);

Зробити Натана -> (Viktor);

{
    Зробити інгредієнт;
    Зробити стакання інгредієнтів;
    Зробити запікання піци;
    Зробити вкладку запікання;
} -> (Ondar Murad);

Вкладка готовки => {
    Холодильник => {
        Зробити інвентар(шнягу для зберігання інгредієнтів);
        Зробити, щоб можна було брати інгредієнти на крафт;
    } -> (Philipp);

    Система крафтів => {
        ScriptableObject'и для крафтів;
        Перевірка крафтів дивлячись на ScriptableObject'и;
    } -> (Ondar Murad);
};

Гравець => {
    Система фінансів => {
        Зробити гроші;
    } -> (Philipp);

    Система репутації => {
        Зробити віднімання репутації при довго приготованній піці;
        Зробити додавання репутації при вчасно приготованній піці;
        Зробити збільшення шансу на чайові залежно від репутації;
        Зробити віднімання репутації при пропуску клієнтів;
        Зробити збільшення кількості клієнтів залежно від репутації;
    } -> (Viktor);
};

Магазин => {
    Зробити покупку => {
        Покупка дає щось, віднімаючи гроші у гравця;
    };
    [[Optional]] Зробити рекламу -> (Viktor);
} -> (Philipp);

{
    [[Optional]] Базове меню;
    [[Optional]] Анімації;
    Зробити перемикання між вкладками;
    Збереження => {
        Збереження грошей;
        Збереження репутації;
        Збереження змісту холодильника;
    };
} -> (Mykhailo);
